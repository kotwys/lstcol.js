(defun lingvo--find-next-tag (tag-name)
  (lambda ()
    (let* ((query `(((open_tag name: (ident) @name)
                     (:match ,tag-name @name))
                    @tag)
           (nodes (treesit-query-capture 'lingvo query
                                         (point)))))
      ;; return first
      (cl-loop for (capture . node) in nodes
               when (eq capture 'tag)
                 return node))))
